Mongo 的 PHP 类封装

严宇川, 2015.11.13

之前有需求使用 PHP 对 Mongo 进行一些操作, 遂简单的进行了一下封装
包括 CURD, 对表模型的字段补全 & 字段检查
不包括索引处理 (必要性不大)

Mongo.php

<?php
/**
 * Mongo PHP 类封装
 * Ryan 2015.11.13
 */
class LibMongo {   
    protected $mongo;
    protected $activeDb;     
    protected $error;  
    protected static $_instance;

    protected static $mongoConf = array(
        'w' => 1,
            // 安全模式, 返回详细结果
        'multiple' => 1,
            // 批量更新
        'upsert' => 1,
            // 更新或插入
        'justOne' => 0,
            // 只删除一条
    );
    // DB 配置
    protected $mongoDbConf = array(
        'local' => array(
            array(
                'hostnames' => '127.0.0.1:27017',
                'database' => 'admin',
                'username' => 'admin',
                'password' => 'admin'
            ),
        ),
    );

    // Table 模型
    public static $tableModel = array(
        'demo' => array(
            'uid' => array(
                'type' => 'integer',
                'default' => '0',
                'desc' => 'UID',
            ),
            'name' => array(
                'type' => 'string',
                'default' => '',
                'desc' => '名称',
            ),
            'tmCreate' => array(
                'type' => 'integer',
                'default' => 0,
                'desc' => '纪录生成时间',
            ),
        )
    );
    /**
     * 构造函数
     * @param [type]  $mongoServer [数组或字符串]
     * @param boolean $connect     [初始化mongo对象时是否连接, 默认连接]
     * @param boolean $autoBalance [是否自动做负载均衡, 默认是]
     */
    private function __construct($dbConf, $connect = true, $autoBalance = true) {     
        try {           
            if (isset($this->mongoDbConf[$dbConf])) {
                $mongoServer = $this->mongoDbConf[$dbConf];
            } else {
                return false;
            }
            $hostnames = array(); 
            if (count($mongoServer) > 1 && $autoBalance) {     
                shuffle($mongoServer);  
            }
            foreach ($mongoServer as $v) {
                $hostnames[] = $v['hostnames'];
            }
            $username = $mongoServer[0]['username'];  
            $password = $mongoServer[0]['password'];
            $database = $mongoServer[0]['database'];
            $mongoServerStr = 'mongodb://' . $username . ':' . $password . '@' . implode(',', $hostnames) . '/' . $database; 

            $this->activeDb = $database;
            // 如果使用命名空间, 需要对实例化类作顶级命名空间转义
            $this->mongo = new MongoClient($mongoServerStr, array('connect' => $connect));
        } catch (MongoConnectionException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();   
        }     
    }
    /**
     * 单例模式
     */
    public static function getInstance($dbConf) {
        if(!isset(self::$_instance)) {
            self::$_instance = new self($dbConf);
        }
        return self::$_instance;
    }    
    /**
     * 简单的模型检查
     * 提供字段补全, String & Int 转换
     */
    protected function modelCheck($table, $param, $type = 'insert') {
        // 如果有表模型定义
        if (isset($this->tableModel[$table])) {
            $tModel = $this->tableModel[$table];

            if ($type == 'insert') {
                // 插入数据时作字段补充
                foreach ($tModel as $k => $v) {
                    // 如果有字段未定义, 且模型中有默认值
                    if (!isset($param[$k]) && isset($v['default'])) {
                        $param[$k] = $v['default'];
                    }
                }
            }
            // 字段检查
            foreach ($param as $k => $v) {
                if (isset($tModel[$k])) {
                    if (!$v && isset($tModel[$k]['default'])) {
                        $v = $tModel[$k]['default'];
                    }
                    switch ($tModel[$k]['type']) {
                        case 'integer' :
                            $param[$k] = intval($v);
                            break;
                        case 'string' :
                            $param[$k] = strval($v);
                            break;
                        default :
                            break;
                    }
                } else {
                    unset($param[$k]);
                }
            }
        }
        return $param;
    }

    /**
     * 选择数据库
     */
    public function selectDb($dbname) {     
        $this->activeDb = $dbname;
        return true;     
    }

    /**
     * 集合查询
     * @param  [type] $table   [集合名称]
     * @param  array  $queryC  [查询条件]
     * @param  array  $resultC [过滤条件]
     * @param  array  $fieldsC [查询字段]
     * @return [type]          [查询结果]
     */
    public function find($table, $queryC = array(), $fieldsC = array(), $resultC = array()) {
        try {     
            $result = array();

            $dbname = $this->activeDb;     
            $cursor = $this->mongo->$dbname->$table->find($queryC);  
            if ($fieldsC) {
                $cursor->fields($fieldsC);
            }   
            if (isset($resultC['start'])) {     
                $cursor->skip($resultC['start']);     
            }     
            if (isset($resultC['limit'])) {     
                $cursor->limit($resultC['limit']);     
            }     
            if (isset($resultC['sort'])) {     
                $cursor->sort($resultC['sort']);     
            } 
            while ($cursor->hasNext()) {     
                $result[] = $cursor->getNext();     
            }     
            return $result;
        } catch (MongoConnectionException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();   
        } catch (MongoCursorTimeoutException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();      
        }          
    } 
    /**
     * 集合查询, 只查询一条纪录
     * @param  [type] $table     [集合名称]
     * @param  [type] $queryC    [查询条件]
     * @param  array  $fieldsC   [查询字段]
     * @return [type]            [查询结果]
     */
    public function findOne($table, $queryC = array(), $fieldsC = array()) {   
        try{    
            $dbname = $this->activeDb;   
            return $this->mongo->$dbname->$table->findOne($queryC, $fieldsC); 
        } catch (MongoCursorException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();     
        }  
    }  

    /**
     * 查询集合的纪录数
     * @param  [type] $table [集合名称]
     * @return [type]        [纪录总数]
     */
    public function count($table) { 
        try{
            $dbname = $this->activeDb; 
            return $this->mongo->$dbname->$table->count();  
        } catch (MongoCursorException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();     
        }  
    }

    /**
     * 插入记录
     * @param  [type] $table  [集合名称]
     * @param  [type] $record [插入数据]
     * @return [type]         [插入结果]
     */
    public function insert($table, $param) {
        $dbname = $this->activeDb;     
        try {     
            // 模型检查
            $param = $this->modelCheck($table, $param, 'insert');

            $result = $this->mongo->$dbname->$table->insert($param, array('w' => self::$mongoConf['w']));
            if ($result['ok']) {
                return $param['_id'];
            } else {
                return false;  
            }   
        } catch (MongoCursorException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();     
        }     
    }        

    /**
     * 更新纪录
     * @param  [type] $table   [集合名称]
     * @param  [type] $queryC  [更新条件]
     * @param  [type] $param   [更新数据]
     * @param  array  $options [更新选项]
     * @return [type]          [更新结果]
     */
    public function update($table, $queryC, $param, $options = array()) {     
        $dbname = $this->activeDb;     

        // 安全模式
        $options['w'] = isset($options['w']) ? $options['w'] : self::$mongoConf['w']; 

        // 批量更新
        $options['multiple'] = isset($options['multiple']) ? $options['multiple'] : self::$mongoConf['multiple'];

        // 更新或插入  
        $options['upsert'] = isset($options['upsert']) ? $options['upsert'] : self::$mongoConf['upsert'];  

        if (isset($options['multiple']) && $options['multiple'] == 0) {     
            $options['multiple'] = 0;          
        }     
        try {     
            // 模型检查
            $param = $this->modelCheck($table, $param, 'update');

            $updateObj = array(
                '$set' => $param
            );
            $result = $this->mongo->$dbname->$table->update($queryC, $updateObj, $options);
            if ($result['ok']) {
                return true;
            } else {
                return false;  
            }   
        } catch (MongoCursorException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();    
        }
    }     

    public function delete($table, $queryC, $options = array()){     
        $dbname = $this->activeDb;
        // 安全模式
        $options['w'] = isset($options['w']) ? $options['w'] : self::$mongoConf['w'];     
        // 删除单条模式
        $options['justOne'] = isset($options['justOne']) ? $options['justOne'] : self::$mongoConf['justOne'];   
        try {     
            $result = $this->mongo->$dbname->$table->remove($queryC, $options);  
            if ($result['ok']) {
                return true;
            } else {
                return false;  
            }   
        } catch (MongoCursorException $e) {     
            $this->error = $e->getMessage();     
            return $this->error();      
        }          
    }    

    /**
     * 获取当前错误信息
     */
    public function getError() {     
        return $this->error;     
    }     

    /**
     * 错误封装返回
     */
    private function error() {
        return false;
        // 是否返回错误
        // return array(
        //     'result' => false,
        //     'error' => $this->error,
        // );
    }
}
