LESS 笔记

简单记录一下LESS 笔记!http://less.bootcss.com

安装

npm install -g less

编译

//执行 
lessc styles.less  

//编译为 CSS 文件  
lessc styles.less > styles.css  
    -x    压缩

开发调试

<link rel="stylesheet/less" type="text/css" href="styles.less" />
<script src="./less/less.js" type="text/javascript"></script>
less.js : https://github.com/less/less.js/archive/master.zip

在线编辑器

  • http://less2css.org/

语法规则

变量

//变量
@blue: #5B83AD;

//选择器
@selector: header;

//URLs
@images: "../img";

//属性
@property: color;

//DEMO
.@{selector}{
    @{property}: @blue;
    background: url("@{images}/bg.png");
}

运算

@base: 10%;
@filler: @base * 2 + @base /2 - 5%;

混合

//.red(), 作为函数使用的CSS, 不表现在样式表中
//.red.inner <=> .red.inner() <=> .red > .inner <=> .red > .inner()

.red() {
    color: red;
    .inner {
        font-size: 12px;
    }
}
#blue-bg {
    background: blue;
}
.border-radius(@radius: 5px) {
    -webkit-border-radius: @radius;
    -moz-border-radius: @radius;
    border-radius: @radius;    
}
.selector {
      .red;
      #blue-bg;
      .red.inner;
      .border-radius(4px);
}

引用

@import "foo";      // foo.less is imported
@import "foo.less"; // foo.less is imported
@import "foo.php";  // foo.php imported as a less file
@import "foo.css";  // statement left in place, as-is

条件

@my-option: false;
.my-optional-style() when (@my-option = true) {
  button {
    color: white;
  }
}
.my-optional-style();

循环

.generate-columns(4);
.generate-columns(@n, @i: 1) when (@i =< @n) {
  .column-@{i} {
    width: (@i * 100% / @n);
  }
  .generate-columns(@n, (@i + 1));
}

合并

.mixin() {
      color: #222222;
}
.myclass {
      .mixin();
      color+: #111111;
}

父选择符 &

//.button:hover, .button-ok, .header .button
.button{
      color: blue;
      &:hover {
        color: green;
      }
      &-ok {
          color: red;
      }
      .header & {
          color: black;
      }
}

函数

杂项函数

//color 
//解析颜色，将代表颜色的字符串转换为颜色值
.demo {
    color: color("#aaa");
}

//convert
//将数字从一种单位转换到另一种单位。

//data-uri
//将资源内联进样式表

//DEMO
.demo {
    background: data-uri('image/jpeg;base64', './logo.png');
}

字符串函数

//escape
//对字符串中的特殊字符做 URL-encoding 编码。
//这些字符不会被编码：,, /, ?, @, &, +, ', ~, ! and $。
//被编码的字符是：\<space\>, #, ^, (, ), {, }, |, :, >, <, ;, ], [ and =。
escape('a=1')

//% 格式化
//此函数 %(string, arguments ...) 用于格式化字符串。
//第一个参数是一个包含占位符的字符串。占位符以百分号 % 开头，后面跟着字母 s、S、d、D、a 或 A。后续的参数用于替换这些占位符。如果你需要输出百分号，可以多用一个百分号来转义 %%。
.demo {
    format-s: %("repetitions: %s file: %s", 1 + 2, "directory/file.less");
}

//replace
//用一个字符串替换一段文本。
@demo: replace("Hello, Mars?", "Mars\?", "Earth!");

列表函数

//length
//返回列表中元素的个数。
//extract
//返回列表中指定位置的元素。

@list: "banana", "tomato", "potato", "peach";
@length: length(@list);
.demo {
    width: @length;
    height: extract(@list, 3);
}

数学函数

//ceil
//向上取整。

//floor
//向下取整。

//percentage
//将浮点数转换为百分比字符串。

//round
//四舍五入取整。

//sqrt
//计算一个数的平方根，并原样保持单位。

//abs
//计算数字的绝对值，并原样保持单位。

//sin
//正弦函数。处理时会将没有单位的数字认为是弧度值。

//asin
//反正弦函数。返回以弧度为单位的角度，区间在 -PI/2 到 PI/2之间。

//cos
//余弦函数。

//acos
//反余弦函数。，区间在 0 到 PI之间。

//tan
//正切函数。

//atan
//反正切函数。

//pi
//返回圆周率 π (pi);

//pow
//设第一个参数为A，第二个参数为B，返回A的B次方。

//mod
//返回第一个参数对第二参数取余的结果。

//min
//返回一系列值中最小的那个。
//max
//返回一系列值中最大的那个。

@num: 2.41;
@num2: 11;
@num3: 3;

.demo {
    ceil: ceil(@num);
    floor: floor(@num);
    percentage: percentage(@num);
    rount: round(@num);
    sqrt: sqrt(@num);
    abs: abs(@num);
    pi: pi();
    pow: pow(@num2, @num3);
    mod: mod(@num2, @num3);
    min: min(@num2, @num3);
    max: max(@num2, @num3);
}

类型函数

//返回 true OR false 

//isnumber
//isstring
//iscolor
//iskeyword
//isurl
//ispixel
//isem
//ispercentage
//isunit

.demo{
    isnumber: isnumber(2);
    isstring: isstring("hello");
    iscolor: iscolor(#000000);
    iskeyword: iskeyword(keyword);
    isurl: isurl(url(...));
    ispixel: ispixel(56px);
    isem: isem(7.8em);
    ispercentage: ispercentage(7.8%);
    isunit: isunit(11px, px);
}

颜色定义函数

//rgb
//通过十进制红色、绿色、蓝色三种值 (RGB) 创建不透明的颜色对象。

//rgba
//通过十进制红色、绿色、蓝色，以及 alpha 四种值 (RGBA) 创建带alpha透明的颜色对象。

//hsl
//通过色相 (hue)，饱和度 (saturation)，亮度 (lightness) 三种值 (HSL) 创建不透明的颜色对象。
//hsla

//从颜色对象中提取颜色通道值。
//red
//green
//blue

.demo {
    rgb: rgb(90, 129, 32);
    rgba: rgba(90, 129, 32, 0.5);
    hsl: hsl(90, 100%, 50%);
    hsla: hsla(90, 100%, 50%, 0.5);
    red: red(rgb(10, 20, 30));
    green: green(rgb(10, 20, 30));
    blue: blue(rgb(10, 20, 30));

}

颜色操作函数

//multiply
//分别将两种颜色的红绿蓝 (RGB) 三种值做乘法运算，然后再除以 255，输出结果是更深的颜色。（译注：对应Photoshop中的“变暗/正片叠底”。）
//screen
//与 multiply 函数效果相反，输出结果是更亮的颜色。（译注：对应Photoshop中的“变亮/滤色”。）
//overlay
//结合 multiply 与 screen 两个函数的效果，令浅的颜色变得更浅，深的颜色变得更深。（译注：对应 Photoshop 中的“叠加”。）注意：输出结果由第一个颜色参数决定。



.demo {
    multiply: multiply(#ff6600, #000000);
}
